\usepackage{xifthen}
\makeatletter
\def\@lecture{}%
\newcommand*{\lecture}[3]{
    \ifthenelse{\isempty{#3}}{%
        \def\@lecture{Lecture #1}%
    }{%
        \def\@lecture{Lecture #1: #3}%
    }%
    \subsection*{\@lecture}
    \marginpar{\raggedright\small\textsf{#2}}%
    \vskip 6pt
}

  \setlength\abovedisplayshortskip{1pt}%
  \setlength\belowdisplayshortskip{1pt}%
}
